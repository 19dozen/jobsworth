all_custom_attribute_values() click to toggle source

Returns an array of the custom attributes values defined for the current object. This will include *at least* the ones defined in available_custom_attributes.

If there are multiple values defined for one attribute type, they will be returned as a nested array. 

 
    # File app/models/custom_attribute_methods.rb, line 22
22:   def all_custom_attribute_values
23:     res = []
24: 
25:     available_custom_attributes.each do |attr|
26:       conds = { :attributable_type => self.class.name, :attributable_id => self.id }
27:       existing = custom_attribute_values.select do |cav|
28:         cav.custom_attribute == attr
29:       end
30: 
31:       if existing.empty?
32:         res << attr.custom_attribute_values.build(conds.merge(:custom_attribute_id => attr.id))
33:       else
34:         res += existing
35:       end
36:     end
37: 
38:     return res
39:   end

set_custom_attribute_values=(params) click to toggle source

Sets up attribute values for this object. Any existing attribute values that are defined, but not passed in will be removed. 

 
    # File app/models/custom_attribute_methods.rb, line 46
46:   def set_custom_attribute_values=(params)
47:     updated = []
48:     existing = custom_attribute_values.clone
49: 
50:     params.each do |values|
51:       attr_id = values[:custom_attribute_id]
52: 
53:       # find an existing value
54:       cav = existing.detect { |v| v.custom_attribute_id == attr_id.to_i }
55:       existing.delete(cav)
56: 
57:       # create a new one if none found
58:       cav ||= custom_attribute_values.build(:custom_attribute_id => attr_id)
59: 
60:       cav.attributes = values
61:       if !new_record?
62:         cav.save
63:       end
64:       updated << cav
65:     end
66: 
67:     # anything still in existing_values hasn't been updated, so delete.
68:     missing = custom_attribute_values - updated
69:     custom_attribute_values.delete(missing)
70:   end
