Builds a new (unsaved) work log for task using the given params params must look like {:work_log=>{…},:comment=>”“} build only if we have :duration or :comment else retur false 

     # File app/models/work_log.rb, line 108
108:   def self.build_work_added_or_comment(task, user, params=nil)
109:     work_log_params=params[:work_log].nil? ? {} : params[:work_log].clone
110:     if (work_log_params and !work_log_params[:duration].blank?) or (params and !params[:comment].blank?)
111:       unless params[:comment].blank?
112:         work_log_params[:body] = params[:comment]
113:         work_log_params[:log_type]=EventLog::TASK_COMMENT
114:         work_log_params[:comment] =true
115:       end
116:       unless work_log_params[:duration].blank?
117:         work_log_params[:duration] = TimeParser.parse_time(user, work_log_params[:duration])
118:         work_log_params[:started_at] = TimeParser.date_from_params(user, work_log_params, :started_at)
119:         work_log_params[:log_type] = EventLog::TASK_WORK_ADDED
120:       else
121:         work_log_params[:duration]=0
122:         work_log_params[:started_at]=Time.now.utc
123:       end
124:       work_log_params[:user]=user
125:       work_log_params[:company]= task.company
126:       work_log_params[:project] = task.project
127:       work_log_params[:customer] = (task.customers.first || task.project.customer)
128:       task.work_logs.build( work_log_params)

130:       return false
